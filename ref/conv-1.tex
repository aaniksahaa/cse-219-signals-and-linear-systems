\begin{frame}{The crossed-pairing rule has a compact formula}

\begin{BlueBox}[
  calloutshadedblue,
  fontupper=\large,
  halign=center
]{}
If
\[
A(z)=\sum_k a[k]z^k,
\qquad
B(z)=\sum_k b[k]z^k,
\]
then the coefficient of \(z^n\) in \(A(z)B(z)\) is
\[
c[n]=\sum_k a[k]\,b[n-k].
\]
\end{BlueBox}

}
